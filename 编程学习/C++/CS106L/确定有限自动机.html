c++
#include <iostream>
#include <set>
#include <map>

using namespace std;

struct DFA
{
    map<pair<int, char>, int> transitions;
    set<int> acceptingStates;
    int startState;
};

bool SimulateDFA(DFA &d, string input)
{
    int currState = d.startState;
    for(string::iterator itr = input.begin(); itr != input.end(); itr++)
    {
        currState = d.transitions[make_pair(currState, *itr)];
    }
    return d.acceptingStates.find(currState) != d.acceptingStates.end();
}

int main()
{
    DFA d;
    pair<int, char> tep = make_pair(0, '0');
    d.transitions[tep] = 1;
    tep = make_pair(0, '1');
    d.transitions[tep] = 2;
    tep = make_pair(1, '0');
    d.transitions[tep] = 0;
    tep = make_pair(1, '1');
    d.transitions[tep] = 3;
    tep = make_pair(2, '0');
    d.transitions[tep] = 3;
    tep = make_pair(2, '1');
    d.transitions[tep] = 0;
    tep = make_pair(3, '0');
    d.transitions[tep] = 2;
    tep = make_pair(3, '1');
    d.transitions[tep] = 1;

    d.acceptingStates.insert(1);
    d.acceptingStates.insert(2);
    d.acceptingStates.insert(3);

    d.startState = 0;

    string s;
    getline(cin, s);

    cout << SimulateDFA(d, s) << endl;

    return 0;
}
